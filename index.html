1. Factory constructor

Factory constructor adalah constructor khusus yang tidak selalu membuat instance baru dari kelasnya. Ia bisa: Mengembalikan instance dari cache (objek yang sudah ada sebelumnya) Mengembalikan instance dari subclass (jenis turunan) Melakukan logika kompleks sebelum membuat objek Berbeda dengan constructor biasa yang selalu membuat objek baru, factory memberi fleksibilitas dalam pengembalian objek. 


// ==========================================
// Nama  : ADRIEL NOFRA MARCELINO
// NIM   : 251410017
// Kelas : SI2A
// Tema  : Factory & Construct - Qurban Style
// ==========================================

class MyClass {

  // Generative constructor
  MyClass();

  // Factory constructor
  factory MyClass.create() {
    return MyClass();
  }
}

void main() {

  var obj1 = MyClass();
  var obj2 = MyClass.create();

  print('Objek 1 berhasil dibuat');
  print('Objek 2 berhasil dibuat');
}

2. Singleton Pattern

Singleton adalah pattern desain kreational (creational pattern) yang memastikan sebuah kelas hanya memiliki satu instance selama program berjalan, serta menyediakan akses global ke instance tersebut. 


// ==========================================
// Nama  : ADRIEL NOFRA MARCELINO
// NIM   : 251410017
// ==========================================

class DatabaseConnection {

  DatabaseConnection._internal() {
    print('Creating database connection...');
  }

  static final DatabaseConnection _instance =
      DatabaseConnection._internal();

  factory DatabaseConnection() {
    return _instance;
  }

  void connect() {
    print('Connected to database');
  }
}

void main() {

  var db1 = DatabaseConnection();
  var db2 = DatabaseConnection();

  print(identical(db1, db2));

  db1.connect();
  db2.connect();
}

3.Object Pool Pattern

Object Pool adalah pattern desain kreational yang menyimpan kumpulan objek siap pakai (pool) untuk digunakan kembali daripada membuat dan menghancurkan objek berulang-ulang. Tujuannya untuk mengoptimalkan performa dan menghemat resource ketika membuat objek itu "mahal" (heavyweight). 


// ==========================================
// Nama  : ADRIEL NOFRA MARCELINO
// NIM   : 251410017
// ==========================================

class ConnectionPool {
  final String host;
  final int port;

  static final Map _cache = {};

  ConnectionPool._(this.host, this.port);

  factory ConnectionPool(String host, int port) {
    var key = '$host:$port';

    if (_cache.containsKey(key)) {
      print('Returning cached connection for $key');
      return _cache[key]!;
    }

    print('Creating new connection for $key');
    var connection = ConnectionPool._(host, port);
    _cache[key] = connection;

    return connection;
  }

  void query(String sql) {
    print('Executing: $sql on $host:$port');
  }
}

void main() {
  var conn1 = ConnectionPool('localhost', 5432);
  conn1.query('SELECT * FROM users');

  print('');

  var conn2 = ConnectionPool('localhost', 5432);
  conn2.query('INSERT INTO products VALUES ("Laptop")');

  print('');

  var conn3 = ConnectionPool('db.example.com', 3306);
  conn3.query('UPDATE orders SET status="Shipped"');

  print('');
  print('conn1 dan conn2 sama? ${identical(conn1, conn2)}');
  print('conn1 dan conn3 sama? ${identical(conn1, conn3)}');
}

5. Factory untuk Configuration

Factory untuk Configuration adalah pola desain yang menggunakan factory pattern untuk membuat, memuat, dan mengelola objek konfigurasi aplikasi. Ini memungkinkan aplikasi untuk Memuat konfigurasi dari berbagai sumber (file, environment, database, API) Mengembalikan format konfigurasi yang berbeda (JSON, YAML, XML, properties) Menerapkan environment-specific coniguration (development, staging, production) Caching dan singleton untuk performa 


// ==========================================
// Nama  : ADRIEL NOFRA MARCELINO
// NIM   : 251410017
// ==========================================

class AppConfig {
  final String environment;
  final String apiUrl;
  final bool debugMode;

  AppConfig._development()
      : environment = 'development',
        apiUrl = 'http://localhost:3000',
        debugMode = true;

  AppConfig._production()
      : environment = 'production',
        apiUrl = 'https://api.example.com',
        debugMode = false;

  AppConfig._staging()
      : environment = 'staging',
        apiUrl = 'https://staging.example.com',
        debugMode = true;

  factory AppConfig.fromEnvironment(String env) {
    switch (env.toLowerCase()) {
      case 'dev':
        return AppConfig._development();
      case 'prod':
        return AppConfig._production();
      case 'stage':
        return AppConfig._staging();
      default:
        return AppConfig._development();
    }
  }

  void printConfig() {
    print('Environment : $environment');
    print('API URL     : $apiUrl');
    print('Debug Mode  : $debugMode');
    print('------------------------');
  }
}

void main() {
  var dev = AppConfig.fromEnvironment('dev');
  var prod = AppConfig.fromEnvironment('prod');
  var stage = AppConfig.fromEnvironment('stage');

  dev.printConfig();
  prod.printConfig();
  stage.printConfig();
}

6. Factory dengan Named Constructors

Factory dengan Named Constructors adalah teknik di mana sebuah kelas memiliki multiple factory constructors dengan nama yang berbeda, masing-masing bertanggung jawab untuk membuat objek dengan cara atau dari sumber yang berbeda. 


// ==========================================
// Nama  : ADRIEL NOFRA MARCELINO
// NIM   : 251410017
// ==========================================

class User {
  final String id;
  final String name;
  final String email;

  User._({
    required this.id,
    required this.name,
    required this.email,
  });

  factory User.fromJson(Map json) {
    return User._(
      id: json['id'],
      name: json['name'],
      email: json['email'],
    );
  }

  factory User.guest() {
    return User._(
      id: '0',
      name: 'Guest',
      email: 'guest@example.com',
    );
  }

  factory User.create(String name, String email) {
    return User._(
      id: '001',
      name: name,
      email: email,
    );
  }

  @override
  String toString() {
    return 'User(id: $id, name: $name, email: $email)';
  }
}

void main() {
  var user = User.fromJson({
    'id': '123',
    'name': 'Alice',
    'email': 'alice@example.com'
  });

  var guest = User.guest();

  var newUser = User.create(
    'Bob',
    'bob@example.com',
  );

  print(user);
  print(guest);
  print(newUser);
}

7. Factory dengan Generic

Factory dengan Generic adalah pola desain di mana factory class atau method menggunakan type parameters (generics) untuk dapat membuat dan mengembalikan objek dari berbagai tipe secara type-safe tanpa kehilangan informasi tipe. Ini memungkinkan: Type safety - Compiler memastikan tipe yang benar Code reuse - Satu factory untuk banyak tipe Flexibility - Mendukung berbagai tipe tanpa casting Compile-time checking - Error terdeteksi saat kompilasi 


// ==========================================
// Nama  : ADRIEL NOFRA MARCELINO
// NIM   : 251410017
// ==========================================

abstract class Response {
  final T? data;
  final String? error;

  Response({this.data, this.error});

  // Factory untuk success
  factory Response.success(T data) = SuccessResponse;

  // Factory untuk error
  factory Response.error(String error) = ErrorResponse;
}

class SuccessResponse extends Response {
  SuccessResponse(T data) : super(data: data);
}

class ErrorResponse extends Response {
  ErrorResponse(String error) : super(error: error);
}

void main() {
  Response success =
      Response.success('Data berhasil dimuat');

  Response error =
      Response.error('Gagal memuat data');

  print('Success Data : ${success.data}');
  print('Error Message : ${error.error}');
}

10.Factory untuk Immutable Objects dengan Cache

Factory untuk Payment Methods adalah pola desain yang digunakan untuk membuat objek pembayaran (credit card, PayPal, crypto, dll) secara terpusat, memungkinkan sistem untuk mendukung berbagai metode pembayaran dengan cara yang konsisten dan mudah ditambahkan metode baru.

// Nama  : ADRIEL NOFRA MARCELINO
// NIM   : 251410017
// ==========================================
class Color {
  final int red;
  final int green;
  final int blue;

  static final Map _cache = {};

  Color._(this.red, this.green, this.blue);

  factory Color.rgb(int red, int green, int blue) {
    if (red < 0 ||
        red > 255 ||
        green < 0 ||
        green > 255 ||
        blue < 0 ||
        blue > 255) {
      throw ArgumentError('RGB values must be 0-255');
    }

    var key = '$red,$green,$blue';

    if (!_cache.containsKey(key)) {
      _cache[key] = Color._(red, green, blue);
    }

    return _cache[key]!;
  }

  factory Color.red() => Color.rgb(255, 0, 0);
  factory Color.green() => Color.rgb(0, 255, 0);
  factory Color.blue() => Color.rgb(0, 0, 255);
  factory Color.white() => Color.rgb(255, 255, 255);
  factory Color.black() => Color.rgb(0, 0, 0);

  @override
  String toString() => 'Color($red, $green, $blue)';
}

void main() {
  var red1 = Color.red();
  var red2 = Color.rgb(255, 0, 0);

  print(red1);
  print(red2);

  print(red1 == red2);
}

12.Challenge – AnimalFactory

Class Animal berperan sebagai class abstrak yang menjadi induk dari beberapa jenis hewan. Berdasarkan parameter yang diberikan, factory constructor akan membuat objek Cat, Dog, atau Bird. Program menerapkan cache sehingga objek hewan dengan atribut yang sama dapat digunakan kembali tanpa membuat objek baru. 


// ==========================================
// Nama  : ADRIEL NOFRA MARCELINO
// NIM   : 251410017
// ==========================================
abstract class Animal {
  final String name;

  Animal(this.name);

  void sound();

  static final Map _cache = {};

  factory Animal.create(
    String type,
    String name,
  ) {
    if (name.isEmpty) {
      throw ArgumentError(
          'Nama hewan tidak boleh kosong');
    }

    String key = '$type-$name';

    if (_cache.containsKey(key)) {
      return _cache[key]!;
    }

    Animal animal;

    switch (type.toLowerCase()) {
      case 'cat':
        animal = Cat(name);
        break;

      case 'dog':
        animal = Dog(name);
        break;

      case 'bird':
        animal = Bird(name);
        break;

      default:
        throw ArgumentError(
            'Jenis hewan tidak dikenal');
    }

    _cache[key] = animal;

    return animal;
  }
}

class Cat extends Animal {
  Cat(String name) : super(name);

  @override
  void sound() {
    print('$name : Meow');
  }
}

class Dog extends Animal {
  Dog(String name) : super(name);

  @override
  void sound() {
    print('$name : Woof');
  }
}

class Bird extends Animal {
  Bird(String name) : super(name);

  @override
  void sound() {
    print('$name : Tweet');
  }
}

void main() {
  var cat1 = Animal.create(
    'cat',
    'Milo',
  );

  var dog1 = Animal.create(
    'dog',
    'Buddy',
  );

  var bird1 = Animal.create(
    'bird',
    'Rio',
  );

  var cat2 = Animal.create(
    'cat',
    'Milo',
  );

  cat1.sound();
  dog1.sound();
  bird1.sound();

  print('Cache bekerja: ${cat1 == cat2}');
}
